def loadChemin():
    directions = []
    xmldoc = minidom.parse("chemin.xml")
    if xmldoc.toxml() == "":
        popup("erreur chargement chemin")
        return
    maps = xmldoc.getElementsByTagName("map")
    for map in maps:
        direction = map.childNodes[1].firstChild.nodeValue
        directions.append(direction)
    popup("Chemin charg√©")
    return directions
